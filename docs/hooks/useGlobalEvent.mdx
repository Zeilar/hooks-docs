```tsx
import { useGlobalEvent } from "@zeilar/hooks";
import React, { useState } from "react";

function UseGlobalEvent() {
	const [key, setKey] = useState<string>();
	useGlobalEvent("keydown", e => setKey(e.key));
	return <pre>{key ?? "No key pressed."}</pre>;
}
```
