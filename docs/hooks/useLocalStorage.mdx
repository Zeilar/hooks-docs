You can find this value in `DevTools -> Application -> Storage (Local Storage) -> https://hooks.angelin.dev -> useLocalStorage`

The input value will be persist until removed.

<UseLocalStorageDemo />

```tsx
import TextField from "@mui/material/TextField";
import { useLocalStorage } from "use-ful-hooks-ts";
import React from "react";

function UseLocalStorageDemo() {
	const [data, setData] = useLocalStorage<string>(
		"useLocalStorage",
		"Hello world"
	);
	return (
		<pre>
			<TextField
				label="Value"
				value={data}
				onChange={e => setData(e.target.value)}
				variant="outlined"
				name="value"
				required
				autoFocus
			/>
		</pre>
	);
}
```
