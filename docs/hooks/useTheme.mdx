```tsx
import Switch from "@mui/material/Switch";
import { useTheme } from "use-ful-hooks-ts";
import React, { useEffect } from "react";

function UseThemeDemo() {
	const [theme, setTheme, toggleTheme] = useTheme("dark");

	useEffect(() => {
		// Changes the data-theme attribute in the <html> element
		document.documentElement.dataset.theme = theme;
	}, [theme]);

	return (
		<pre>
			<p>Theme: {theme}</p>
			<Switch checked={theme === "light"} onChange={toggleTheme} />
		</pre>
	);
}
```
